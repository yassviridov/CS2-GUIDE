⚙️ готовый конфиг

⌨️

autoexec.cfg

// основные бинды
bind "z" "slot7"           // дым
bind "x" "slot8"           // светошумовая  
bind "c" "slot6"           // осколочная
bind "v" "slot10"          // коктейль

// отключение голосового чата
bind "t" "toggle voice_modenable 1 0"

// jumpthrow (для смоков)
alias +jumpthrow "+jump;-attack"
alias -jumpthrow "-jump"
bind "space" +jumpthrow

// сброс бомбы
bind "b" "use weapon_knife; use weapon_c4; drop; slot1"

// громкость шагов
alias +incvol "incrementvar volume 0 2 0.5; +sprint"
alias -incvol "incrementvar volume 0 2 -0.5; -sprint"
bind "shift" +incvol

// очистка крови
bind "mouse3" "r_cleardecals"
📁 сохрани в \cs2\cfg\autoexec.cfg
